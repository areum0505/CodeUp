### 1705 : 369 게임

기은이가 369게임에서 자신의 차례가 되었을때 무엇을 내야 하는지 알아내는 프로그램을 작성하시오.

``` c
#include <stdio.h>

int main() {
	int n, r, k = 1, temp = 0;
	scanf("%d", &n);
	r = n;
	temp = n % 10;

	if (temp == 3 || temp == 6 || temp == 9) {
		k = 0;
		printf("K");
	}
	n /= 10;
	temp = n % 10;
	if (temp == 3 || temp == 6 || temp == 9) {
		k = 0;
		printf("K");
	}
	n /= 10;
	temp = n % 10;
	if (temp == 3 || temp == 6 || temp == 9) {
		k = 0;
		printf("K");
	}

	if (k != 0)
		printf("%d", r);
		
	return 0;
}
```
